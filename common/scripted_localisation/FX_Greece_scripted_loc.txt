	name = GetGoldenDawnInfluence
	text = {
		trigger = {
			AND = {

			}
		}
		localization_key = golden_dawn_influence_20
	}

		localization_key = golden_dawn_influence_40

				check_variable = { golden_dawn_influence > 40 }
			}

		localization_key = golden_dawn_influence_60
	}
	text = {
