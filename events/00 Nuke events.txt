# Hidden event to set flag on nuke dropped

	id = nuke_dropped.0

	major = yes

	is_triggered_only = yes

	hidden = yes

	fire_only_once = yes

	immediate = {
		news_event = {
			id = nuke_dropped.1
			days = 1
		}
	}

	option = {
		#name = dummy_nuke_option
	}
}

# Nuclear Attack PLACEHOLDER
news_event = {
	id = nuke_dropped.1
	title = nuke_dropped.1.t
	desc = nuke_dropped.1.d
	picture = GFX_news_event_nuke

	major = yes

	is_triggered_only = yes

	trigger = {
		NOT = {
			FROM.FROM = {
				OR = {
					state = 2 #Latium
					state = 16 #Paris
					state = 37 #Copenhagen
					state = 64 #Berlin
					state = 126 #Greater London
					state = 141 #Stockholm
					state = 219 #Moscow
					state = 282 #Tokyo
					state = 289 #Siam
					state = 361 #Washington D.C.
					state = 528 #Nagasaki
					state = 529 #Hiroshima
					state = 531 #Kyoto
				}
			}
		}
	}

	option = {
		name = nuke_dropped.1.a
		trigger = {
			NOT = { owns_state = FROM.FROM }
		}
	}

	option = {
		name = nuke_dropped.1.b
		trigger = {
			owns_state = FROM.FROM
		}
	}
}
