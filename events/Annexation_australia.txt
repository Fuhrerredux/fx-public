	option = { # Puppet

		trigger = {
			NOT = { 
				AND = {
					tag = ENG
					has_country_flag = federation
				}
			}
		}
		ai_chance = {
			base = 100
		}
		set_temp_variable = { tag_to_release = AST }
		release_targeted_tag = yes
		if = {
			limit = { tag = ENG }
			AST = {
				set_cosmetic_tag = AST_ENG

			}
		}
		else = {
			AST = { AST_become_republic = yes }
		}
	}

	option = { # Imperial Federation
		log = "[GetLogInfo]: event annex.5001 option annex.integrate"
		name = annex.integrate
		trigger = {
			tag = ENG
			has_country_flag = federation
		}
		ai_chance = {
			base = 100
		}
		set_country_flag = AST_CONQ
		for_each_scope_loop = {
			array = AST.core_states
			add_core_of = ROOT
		}
	}

	option = { # Occupy
